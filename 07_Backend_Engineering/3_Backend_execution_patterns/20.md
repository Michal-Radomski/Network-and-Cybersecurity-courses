Here is a complete, real-world example of the **Idempotency Key Pattern** implemented using an Express.js backend (with Redis
for caching) and a React.js frontend.

In this scenario, we ensure that a user clicking a "Pay $100" button multiple times by accident will only result in a single
transaction.

---

## 1. The Backend (Express.js + Redis)

The backend uses Redis to store the state and response of each `Idempotency-Key`.

If a request with the same key arrives while the first one is processing, it blocks it (`409 Conflict`). If it arrives after
completion, it immediately returns the cached response (`200 OK`).

```javascript
// server.js
const express = require("express");
const redis = require("redis");
const { v4: uuidv4 } = require("uuid");

const app = express();
app.use(express.json());

// Initialize Redis Client
const redisClient = redis.createClient();
redisClient.connect().catch(console.error);

// Mock function representing a slow, heavy database/payment gateway operation
const fakePaymentGateway = () => new Promise((resolve) => setTimeout(resolve, 3000));

app.post("/api/checkout", async (req, res) => {
  const idempotencyKey = req.headers["idempotency-key"];
  const { amount, userId } = req.body;

  // 1. Fallback if frontend forgot to send a key
  if (!idempotencyKey) {
    return res.status(400).json({ error: "Missing Idempotency-Key header" });
  }

  const redisKey = `idempotency:${userId}:${idempotencyKey}`;

  try {
    // 2. Check if this key already exists in Redis
    const cachedRecord = await redisClient.get(redisKey);

    if (cachedRecord) {
      const parsedRecord = JSON.parse(cachedRecord);

      // Scenario A: The initial request is still processing
      if (parsedRecord.status === "STARTED") {
        return res.status(409).json({
          error: "Your request is already being processed. Please wait.",
        });
      }

      // Scenario B: The request already finished successfully. Return cached data.
      if (parsedRecord.status === "COMPLETED") {
        return res.status(200).json({
          ...parsedRecord.response,
          _cached: true, // flag just to prove it came from cache
        });
      }
    }

    // 3. First time seeing this key. Lock it in Redis as 'STARTED'
    // We set a 5-minute TTL so if the server crashes, the user isn't locked out forever.
    await redisClient.set(redisKey, JSON.stringify({ status: "STARTED" }), { EX: 300 });

    // 4. Execute the critical business logic (e.g., charge credit card)
    await fakePaymentGateway();

    const responsePayload = {
      success: true,
      transactionId: `tx_${uuidv4().slice(0, 8)}`,
      amount,
      message: `Successfully charged $${amount}`,
    };

    // 5. Save the final successful response in Redis (Cache for 24 hours)
    await redisClient.set(
      redisKey,
      JSON.stringify({
        status: "COMPLETED",
        response: responsePayload,
      }),
      { EX: 86400 }
    );

    // 6. Respond to the client
    return res.status(200).json(responsePayload);
  } catch (error) {
    console.error(error);
    // On failure, remove the lock so the user can actually try again safely
    await redisClient.del(redisKey);
    return res.status(500).json({ error: "Internal Server Error" });
  }
});

app.listen(5000, () => console.log("Backend listening on port 5000"));
```

---

## 2. The Frontend (React.js)

The golden rule for the frontend is: **The `Idempotency-Key` must be tied to the component's lifecycle or state, NOT
generated inside the submit function.** If you generate it _inside_ the submit function, every retry click creates a
completely brand-new key, which entirely defeats the point of backend tracking.

```jsx
// CheckoutButton.jsx
import React, { useState, useEffect } from "react";

export default function CheckoutButton() {
  const [idempotencyKey, setIdempotencyKey] = useState("");
  const [loading, setLoading] = useState(false);
  const [statusMessage, setStatusMessage] = useState("");

  // 1. Generate the unique key when the page/modal mounts.
  // It stays exactly the same no matter how many times the user clicks the button.
  useEffect(() => {
    const uniqueKey = crypto.randomUUID(); // Native browser API for generating UUIDs
    setIdempotencyKey(uniqueKey);
  }, []);

  const handlePayment = async () => {
    setLoading(true);
    setStatusMessage("Processing payment...");

    try {
      const response = await fetch("/api/checkout", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Idempotency-Key": idempotencyKey, // 2. Pass the key in the headers
        },
        body: JSON.stringify({
          amount: 100,
          userId: "user_99",
        }),
      });

      const data = await response.json();

      if (response.status === 200) {
        setStatusMessage(`Success! TxID: ${data.transactionId} ${data._cached ? "(Cached)" : ""}`);

        // 3. Regenerate a brand-new key ONLY after an absolute success,
        // so they can make a completely separate purchase if they want to.
        setIdempotencyKey(crypto.randomUUID());
      } else if (response.status === 409) {
        // Handle concurrent clicking warning gracefully
        setStatusMessage(data.error);
      } else {
        setStatusMessage("Payment failed. Please try again.");
      }
    } catch (error) {
      // Network drops land here.
      // If the user clicks again, the EXACT same idempotencyKey is sent.
      setStatusMessage("Network issue. Retrying connection...");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div style={{ padding: "20px", fontFamily: "sans-serif" }}>
      <h2>Your Cart: 1 Jacket ($100)</h2>
      <button
        onClick={handlePayment}
        disabled={loading}
        style={{ padding: "10px 20px", fontSize: "16px", cursor: "pointer" }}
      >
        {loading ? "Submitting..." : "Pay $100"}
      </button>

      <p>
        <strong>Status:</strong> {statusMessage}
      </p>
      <small style={{ color: "#888" }}>Current Idempotency Key: {idempotencyKey}</small>
    </div>
  );
}
```

---

## How to test this example:

1. Fire up the React app and click the button once.
2. The UI will show "Processing payment..." for 3 seconds.
3. While it is processing, click the button a second or third time. You will see the UI immediately change to **"Your request
   is already being processed. Please wait."** because the backend caught the duplicate key via Redis and returned a `409`.
4. Once the original 3 seconds wrap up, any subsequent accidental clicks will hit the `200` block and quickly return the
   cached message **"(Cached)"** without running the checkout process again.
